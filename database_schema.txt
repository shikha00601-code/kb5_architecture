ShopStream Database Schema
PostgreSQL 15

ORDER-SERVICE DATABASE (orderdb):

orders table:
  - id: UUID (PK)
  - customer_id: UUID (indexed)
  - status: VARCHAR (CREATED, PAYMENT_PENDING, CONFIRMED, SHIPPED, 
            DELIVERED, CANCELLATION_PENDING, CANCELLED)
  - total_amount: DECIMAL(12,2)
  - original_currency: VARCHAR(3)
  - exchange_rate_applied: DECIMAL(10,6)
  - usd_equivalent: DECIMAL(12,2)
  - metadata: JSONB (flexible product-specific data)
  - created_at: TIMESTAMPTZ
  - updated_at: TIMESTAMPTZ
  - deleted_at: TIMESTAMPTZ (soft delete)
  
  Indexes:
  - idx_orders_customer_created: (customer_id, created_at DESC)
  - idx_orders_status: (status)
  - idx_orders_deleted: (deleted_at) WHERE deleted_at IS NULL

order_items table:
  - id: UUID (PK)
  - order_id: UUID (FK -> orders.id)
  - product_id: UUID
  - product_name: VARCHAR
  - quantity: INTEGER
  - unit_price: DECIMAL(10,2)
  - subtotal: DECIMAL(10,2)

processed_events table (idempotency):
  - event_id: VARCHAR (PK)
  - processed_at: TIMESTAMPTZ
  - event_type: VARCHAR

PAYMENT-SERVICE DATABASE (paymentdb):

payments table:
  - id: UUID (PK)
  - order_id: UUID (indexed, unique)
  - stripe_payment_intent_id: VARCHAR
  - amount: DECIMAL(12,2)
  - currency: VARCHAR(3)
  - status: VARCHAR (PENDING, COMPLETED, FAILED, REFUNDED)
  - created_at: TIMESTAMPTZ
  - completed_at: TIMESTAMPTZ

INVENTORY-SERVICE DATABASE (inventorydb):

inventory table:
  - product_id: UUID (PK)
  - product_name: VARCHAR
  - available_quantity: INTEGER
  - reserved_quantity: INTEGER
  - total_quantity: INTEGER (computed: available + reserved)
  - updated_at: TIMESTAMPTZ

reservations table:
  - id: UUID (PK)
  - order_id: UUID
  - product_id: UUID
  - quantity: INTEGER
  - reserved_at: TIMESTAMPTZ
  - expires_at: TIMESTAMPTZ
  - status: VARCHAR (ACTIVE, CONFIRMED, RELEASED, EXPIRED)
